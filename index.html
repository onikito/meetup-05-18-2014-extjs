Route Handling

        'user/:id' : {
            before  : 'onBeforeUser',
            action  : 'onUser'
        }
    },
    onBeforeUser : function(id, action) {
        Ext.Ajax.request({
            url     : '/security/user/' + id,
            success : function() {
                action.resume();
            }
        });
